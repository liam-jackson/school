function p = p_x_giv_c(att_mesh, att_vals, sigma)

k = 1/(sigma*sqrt(2*pi));
m = length(att_vals);
p = 0;

for instance = 1:length(att_vals)
    p = p + exp((-1/(2*(sigma^2))).*(att_mesh - att_vals(instance)).^2);
end

p = p*k/m;

end
