train_labels_ohenc = np.zeros([10, len(train_labels)])

    train_labels_ohenc_idx = train_labels[inst] 
    train_labels_ohenc[train_labels_ohenc_idx, inst] = 1

    test_labels_ohenc[test_labels_ohenc_idx, inst] = 1
